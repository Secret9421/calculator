    <div class="container">
        <h1>图形面积计算器</h1>
        <p>请选择要计算的图形：</p >
        
        <div class="option">
            <label>
                <input type="radio" name="shape" value="circle" checked> 圆形
            </label>
            <label style="margin-left: 30px;">
                <input type="radio" name="shape" value="rect"> 矩形
            </label>
        </div>

        <div id="circleInput">
            <label for="radius">圆的半径：</label>
            <input type="number" id="radius" step="any" placeholder="输入半径（支持整数/小数）">
        </div>

        <div id="rectInput" style="display: none;">
            <label for="width">矩形的长：</label>
            <input type="number" id="width" step="any" placeholder="输入长">
            <label for="height">矩形的宽：</label>
            <input type="number" id="height" step="any" placeholder="输入宽">
        </div>

        <button onclick="calculate()">计算面积</button>

        <div id="result" class="result" style="display: none;"></div>
    </div>

    <script>
        const PI = 3.1415926;

        // 切换显示输入框
        document.querySelectorAll('input[name="shape"]').forEach(radio => {
            radio.addEventListener('change', function() {
                document.getElementById('circleInput').style.display = this.value === 'circle' ? 'block' : 'none';
                document.getElementById('rectInput').style.display = this.value === 'rect' ? 'block' : 'none';
                document.getElementById('result').style.display = 'none';
            });
        });

        // 计算函数（复刻C++逻辑）
        function calculate() {
            const shape = document.querySelector('input[name="shape"]:checked').value;
            const resultDiv = document.getElementById('result');
            resultDiv.style.display = 'block';
            resultDiv.className = 'result';

            try {
                if (shape === 'circle') {
                    const r = parseFloat(document.getElementById('radius').value);
                    if (isNaN(r) || r <= 0) throw "请输入有效的正数半径！";
                    const area = PI * r * r;
                    resultDiv.textContent = `圆的面积为：${area.toFixed(2)}`;
                } else {
                    const w = parseFloat(document.getElementById('width').value);
                    const h = parseFloat(document.getElementById('height').value);
                    if (isNaN(w) || isNaN(h) || w <= 0 || h <= 0) throw "请输入有效的正数长和宽！";
                    const area = w * h;
                    resultDiv.textContent = `矩形的面积为：${area.toFixed(2)}`;
                }
            } catch (e) {
                resultDiv.textContent = e;
                resultDiv.className = 'result error';
            }
        }
    </script>
